# 下载

下载模板、脚手架等

# 使用说明

// 安装
npm i -g @hnrfqj/rfqj-cli

// 使用
rfqj clone boilerplateName [-p=./]
 - boilerplateName： wxminiapp、opAdmin、vuePage、nuxtPage
 - [-p=./] 保存路径，默认./，非必传


// 下载微信小程序脚手架
rfqj clone wxminiapp -p=./

// 下载运营后台脚手架
rfqj clone opAdmin

// 下载 vue 单页代码片段
rfqj clone vuePage

// 下载 nuxt 单页代码片段
rfqj clone nuxtPage
